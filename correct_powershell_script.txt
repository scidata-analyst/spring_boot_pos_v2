# ==============================
# Helper function to ensure folder exists
# ==============================
function Ensure-Folder($path){
    if (!(Test-Path $path)) { New-Item -Path $path -ItemType Directory | Out-Null }
}

# Base path for jobs
$jobsBasePath = "D:\OWN\spring_boot_pos_v2\POS_APP\src\main\java\com\example\pos\jobs"

# Ensure base jobs folder exists
Ensure-Folder $jobsBasePath

# Schema (modules and entities)
$schema = @{
    "Accounting" = @("ApAr", "CashRegister", "Reconciliation", "TaxReports")
    "Core" = @("Dashboard", "PosTerminal")
    "Customers" = @("Customers", "LoyaltyPoints", "Memberships", "PurchaseHistory")
    "Hardware" = @("CashDrawer", "Devices", "QrMobile")
    "Loyalty" = @("PointsConfig", "Retention", "TierManagement")
    "Payment" = @("ApAr", "CashRegister", "Reconciliation", "TaxReports")
    "Products" = @("Barcodes", "BatchExpiry", "Catalog", "Categories", "StockMovement")
    "Promos" = @("Coupons", "HappyHour", "Offers")
    "Reports" = @("CustomReport", "PlReport", "SalesReport", "StockReport")
    "Sales" = @("HeldOrders", "Invoices", "Returns", "Transactions")
    "Suppliers" = @("PurchaseOrders", "RestockAlerts", "SupplierPayments", "Suppliers")
    "Users" = @("ActivityLog", "Cashiers", "Roles")
}

foreach ($module in $schema.Keys) {
    $entities = $schema[$module]

    # Create module folder under jobs
    $moduleJobDir = Join-Path $jobsBasePath $module
    Ensure-Folder $moduleJobDir

    foreach ($entity in $entities) {
        $jobFile = Join-Path $moduleJobDir "$entity`Job.java"

        $jobContent = @"
package com.example.pos.jobs.$module;

import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

/**
 * Scheduled job class for $entity.
 * <p>
 * This class is responsible for executing recurring tasks related to $entity.
 * Define job logic inside the execute() method.
 */
@Component
public class ${entity}Job {

    /**
     * Executes the scheduled job for $entity.
     * Runs every hour (customize cron expression as needed).
     */
    @Scheduled(cron = ""0 0 * * * ?"")
    public void execute() {
        // TODO: Implement the job logic for $entity
        System.out.println(""Executing job for $entity..."");
    }
}
"@

        # Write Job file
        $jobContent | Out-File -FilePath $jobFile -Encoding UTF8
        Write-Host "Job created: $module\$entity`Job.java"
    }
}
