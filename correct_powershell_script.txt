$schemaPath = "F:\DASHBOARD_APP\GIT_VERSION\POS\schema.json"

$basePath = "F:\DASHBOARD_APP\GIT_VERSION\POS\POS_APP\src\main\java\com\example\pos\controllers"

function Ensure-Folder($path) {
    if (!(Test-Path $path)) {
        New-Item -Path $path -ItemType Directory | Out-Null
    }
}

$schema = Get-Content $schemaPath | ConvertFrom-Json

foreach ($group in $schema.groups) {

    foreach ($entity in $group.entities) {

        $entityName = $entity.name

        $modulePath = Join-Path $basePath $group.name
        Ensure-Folder $modulePath

        $filePath = Join-Path $modulePath "$entityName`Controller.java"

        $content = @"
package com.example.pos.controllers.$($group.name);

import com.example.pos.dtos.request.$($group.name).${entityName}Request;
import com.example.pos.dtos.response.$($group.name).${entityName}Response;
import com.example.pos.services.$($group.name).${entityName}Service;

import jakarta.validation.Valid;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.List;

/**
 * =====================================================
 * REST Controller: $entityName
 * =====================================================
 *
 * Provides standard REST APIs for $entityName.
 * Uses repository-semantic service methods:
 * all, index, view, create, update, delete.
 */
@RestController
@RequestMapping("/api/$($group.name)/$entityName")
public class ${entityName}Controller {

    @Autowired
    private ${entityName}Service service;

    /**
     * Get all records.
     */
    @GetMapping("/all")
    public List<${entityName}Response> all() {
        return service.all();
    }

    /**
     * Get paginated/index data.
     */
    @GetMapping("/index")
    public List<${entityName}Response> index() {
        return service.index();
    }

    /**
     * View single record.
     */
    @GetMapping("/view/{id}")
    public ${entityName}Response view(@PathVariable Long id) {
        return service.view(id);
    }

    /**
     * Create new record.
     */
    @PostMapping("/create")
    public ${entityName}Response create(@RequestBody @Valid ${entityName}Request request) {
        return service.create(request);
    }

    /**
     * Update existing record.
     */
    @PutMapping("/update/{id}")
    public ${entityName}Response update(@PathVariable Long id,
            @RequestBody @Valid ${entityName}Request request) {
        return service.update(id, request);
    }

    /**
     * Delete record.
     */
    @DeleteMapping("/delete/{id}")
    public String delete(@PathVariable Long id) {
        service.delete(id);
        return "Deleted successfully";
    }
}
"@

        $content | Out-File -FilePath $filePath -Encoding UTF8

        Write-Host "Controller Updated: $group -> $entityName"
    }
}
